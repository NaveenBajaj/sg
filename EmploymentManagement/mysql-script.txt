create table employee (
employeeId varchar(10) PRIMARY KEY, 
firstName varchar(20), 
lastName varchar(20), 
aadharId varchar(30), 
mobileNo varchar(20), 
isPf varchar(5),
pf varchar(30), 
isEsic varchar(5),
esic varchar(30), 
fatherName varchar(20), 
position varchar(10), 
address varchar(100),
salaryBasis varchar(10)
);

create table employee_account (
employeeId varchar(10),
effectiveDate date,
basicSalary varchar(10),
hra varchar(10),
convience varchar(10),
otherAllowance varchar(10),
ratePerDay varchar(10),
PRIMARY KEY (employeeId,effectiveDate),
FOREIGN KEY (employeeId) REFERENCES employee(employeeId)
);

create table salary (
employeeId varchar(10),
salaryDate date,
noOfHours varchar(10),
extraHours varchar(10),
basicSalary varchar(10),
hra varchar(10),
convience varchar(10),
otherAllowance varchar(10),
ratePerDay varchar(10),
leaves varchar(10),
pfAmount varchar(10),
esicAmount varchar(10),
grossSalary varchar(10),
netPay varchar(10),
PRIMARY KEY (employeeId,salaryDate),
FOREIGN KEY (employeeId) REFERENCES employee(employeeId)
);
